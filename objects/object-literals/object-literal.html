Object literals are objects defines using { } and key-value properties

A simple object

const cat = {
  name: "michi",
  age: 4,
}

Can have methods too

const cat = {
  name: "michi",
  age: 4,
  askForFood: function() {
    return "Miaaaau, Meeeow, Gggggh!";
  }
}

So we can call name, age and askForFood on cat, using dot notation or brackets

console.log(cat.name);
console.log(cat.['name']); // -> michi

console.log(cat.age);
console.log(cat.['age']); // -> 4

console.log(cat.askForFood()); // -> Miaaaau, Meeeow, Gggggh!
console.log( cat['askForFood']() ); // Be careful with the syntax!

A property can be a computed (use this only if needed)

let propertyName = 'name';

let book = {
  [propertyName]: 'Ficciones',
}
console.log(book.name); // -> Ficciones

Property and values shorthand -- the next one is a function factory

function createBook(title, author, pages) {
  return { title:title, author: author, pages: pages};
}
// could be written as follows
function createBook2(title, author, pages) {
  return { title, author, pages};
}

How to know if a property exists in an object

If a property does NOT exists and we try to call it, we receive undefined

let house = { 'address': 123, price: 1000000};
console.log(house.rooms); // -> undefined
// we can use the in operator to check, in case the property exists but holds 'undefined' as value
console.log('rooms' in house); // -> false

The for ... in loop

function createBook(title, author, pages) {
  return { title, author, pages};
}

const book = createBook('Ficciones', 'J.L.B', 100);

for( let key in book) {
  console.log(`${key}: ${book[key]}`);
}

How to delete a property?

Use delete object.property

function createBook(title, author, pages, price) {
  return { title, author, pages, price};
}

const book = createBook('Ficciones', 'J.L.B', 100, 200);

for( let key in book) {
  console.log(`${key}: ${book[key]}`);
}

// now delete the price
delete book.price; // -> true

for( let key in book) {
  console.log(`${key}: ${book[key]}`);
}
