Factory functions

A factory function is a function tha returns an object

function person(name, age) { const buildHouse = function() { return `${name} is building a house`; } return {name, age, buildHouse}; } // create an object const bob = person('bob', 23); console.log(bob.buildHouse());

or

// be careful with 'this' const person = function(name, age) { const buildHouse = () => { return `${name} is building a house`; } return {name, age, buildHouse}; }

or

// here againg, 'this' const person = (name, age) => { const buildHouse = () => `${name} is building a house`; return {name, age, buildHouse}; }
